Create the messages table with Relation to the Users Table This table will store the chat messages between users It includes foreign key constraints to ensure data integrity and indexing for faster lookups when loading chat history. CREATE TABLE IF NOT EXISTS messages ( id INT AUTO_INCREMENT PRIMARY KEY, sender_id INT NOT NULL, receiver_id INT NOT NULL, message_text TEXT NOT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, is_read TINYINT(1) DEFAULT 0, -- Foreign Key Constraints: -- This ensures that if a user is deleted, their messages are also handled. -- It also prevents messages from being sent to non-existent user IDs. CONSTRAINT fk_sender FOREIGN KEY (sender_id) REFERENCES users(id) ON DELETE CASCADE, CONSTRAINT fk_receiver FOREIGN KEY (receiver_id) REFERENCES users(id) ON DELETE CASCADE, -- Indexing for faster lookups when loading chat history INDEX (sender_id), INDEX (receiver_id) ); #--. Create the adoptions table to track pet adoptions This table stores which users have adopted which pets and the status of each adoption (pending, approved, completed, rejected) CREATE TABLE IF NOT EXISTS adoptions ( id INT AUTO_INCREMENT PRIMARY KEY, pet_id INT NOT NULL, adopter_id INT NOT NULL, adoption_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP, status ENUM('pending', 'approved', 'completed', 'rejected') DEFAULT 'pending', notes TEXT, FOREIGN KEY (pet_id) REFERENCES pets(id) ON DELETE CASCADE, FOREIGN KEY (adopter_id) REFERENCES users(id) ON DELETE CASCADE, UNIQUE KEY unique_adoption (pet_id, adopter_id), INDEX idx_adopter (adopter_id), INDEX idx_status (status) );
